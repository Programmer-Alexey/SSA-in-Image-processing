\section{Постановка задачи}

\begin{frame}{Постановка задачи}
	\begin{block}{Постановка задачи обнаружения прямых}
		\begin{itemize}
			\item Наблюдаемое изображение: матрица яркостей $\mathbf{M} \in \mathbb{R}^{m \times n}$.

			\item Модель наблюдения:
			\[
			\mathbf{M} = \mathbf{S} + \mathbf{R},
			\]
			где $(\bf{R})_{ij}\stackrel{\text{i.i.d.}}{\sim}\mathcal{N}(0,\sigma^2)$.
			
			\item В рассматриваемом эксперименте компонента сигнала $\mathbf{S}$ такова, что пикселям на $a$-й прямой соответствует константа $c_a>0$, задающая её интенсивность, а вне прямых стоят нули.
			\item Цель: по зашумлённому изображению выделить прямые линии и оценить их параметры.
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}{Преобразование Хафа и предобработка}
	\begin{itemize}
		\item Существует преобразование Хафа, которое решает задачу оценки параметров прямых на изображении.
		\item На практике преобразование Хафа обычно применяется после некоторой предобработки изображения.
		\item В этой работе преобразование Хафа используется как этап финальной оценки параметров после шумоподавления.
	\end{itemize}

	\vspace{0.25cm}
	\begin{block}{Почему возникает этап предобработки}
		\begin{itemize}
			\item Преобразование Хафа ищет максимумы в аккумуляторном массиве, который будет формально определен далее.
			\item Шум порождает ложные активные точки и, следовательно, ложные максимумы.
			\item Поэтому перед преобразованием Хафа изображение сначала подвергают шумоподавлению.
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}{Методы шумоподавления перед преобразованием Хафа}
	\small
	\begin{enumerate}
		\item Рассматриваются четыре стандартных варианта предобработки: \texttt{BM3D}, медианный фильтр как локальная ранговая обработка, фильтр Винера как локальная линейная оценка и \texttt{Quantile $p$} в качестве самого простого алгоритма для сравнения.
		\item Ограничение квантильного фильтра: если одна прямая имеет сильный сигнал, а другая слабый сигнал, сравнимый с шумом, глобальный квантильный порог может сохранить в основном сильную прямую и потерять слабую.
	\end{enumerate}

	\vspace{0.2cm}
	Наряду с ними далее определяется алгоритм \texttt{CSSA ROW-ROW}, основанный на преобразовании Фурье и комплексном SSA.
\end{frame}

\section{Преобразование Хафа}

\begin{frame}{Преобразование Хафа: замечание о параметризации прямой}
	\small
	\begin{itemize}
		\item Часто прямую на плоскости задают в виде графика линейной функции $y = ax + b$.
		\item Для вертикальной прямой $x = c$ коэффициент наклона $a$ не является конечным.
		\item Поэтому в преобразовании Хафа используют перепараметризацию
		\[
		\rho = x \cos \theta + y \sin \theta,
		\]
		где $\rho\in \mathbb{R}$ и $\theta\in[0,\pi]$.
		\item В рассматриваемой задаче изображение задается на дискретной матрице, поэтому считаем $\rho\in [-\rho_{\max}, \rho_{\max}]$, $\rho_{\max}$ можно непосредственно вычислить по матрице.
		
	\end{itemize}
\end{frame}

\begin{frame}{Преобразование Хафа: параметризация $(\rho,\theta)$}
	\begin{figure}
		\centering
		\includegraphics[height=0.58\textheight,keepaspectratio]{images/rt_params.png}
		\caption{Параметризация прямой через $\rho$ и $\theta$}
	\end{figure}
\end{frame}

\begin{frame}{Преобразование Хафа: идея}
	\begin{itemize}
		\item После введения параметризации $(\rho,\theta)$ задачу обнаружения прямой можно перенести в пространство параметров.
		\item Для фиксированной точки изображения $(x_0,y_0)$ множество параметров всех прямых, проходящих через нее, задается соотношением
		\[
		\rho(\theta) = x_0 \cos \theta + y_0 \sin \theta,
		\]
		то есть одной кривой в параметрическом пространстве.
		\item Если точки $(x_1,y_1),\dots,(x_s,y_s)$ лежат на одной прямой, то соответствующие кривые проходят через общую точку $(\rho^\ast,\theta^\ast)$.
		\item Эта общая точка задаёт параметры исходной прямой.
		\item Поэтому задача выделения прямых сводится к поиску точек пересечения таких кривых, а после дискретизации --- к поиску максимумов в аккумуляторном массиве.
	\end{itemize}
\end{frame}

\begin{frame}{Активное множество после предобработки}
	\small
	После предобработки преобразование Хафа применяется не ко всем пикселям, а только к множеству активных пикселей.
	При выбранном нижнем пороге $\varepsilon>0$ оно задается как
	\[
	E_{\varepsilon} = \{(i,j): \widehat{M}_{ij} > \varepsilon\}.
	\]

	\begin{block}{Смысл порога}
		Порог $\varepsilon$ отсеивает слабые положительные остатки, которые могут быть вызваны шумом и иначе участвовали бы в голосовании.
	\end{block}

	\begin{itemize}
		\item Поэтому слабая визуальная выраженность шума не гарантирует малую ошибку по $(\rho,\theta)$.
	\end{itemize}
\end{frame}

\begin{frame}{Аккумуляторный массив}
	\small
	\begin{block}{Определение}
		{\scriptsize
			Пусть $E_{\varepsilon}$ --- множество активных точек изображения,
		$h_\rho>0$ и $h_\theta>0$ --- шаги дискретизации по параметрам $\rho$ и $\theta$.
		Сетка в пространстве параметров задаётся как
		\[
			\rho_a = -\rho_{\max} + a h_\rho,\qquad
			\theta_b = b h_\theta.
		\]
			Тогда аккумуляторный массив $A=(A_{ab})$ определяется формулой
		\[
			A_{ab} =
			\sum_{(x,y)\in E_{\varepsilon}}
			\mathbf{1}\left\{
			\left| \rho_a - \bigl(x\cos\theta_b + y\sin\theta_b\bigr) \right|
			\le \frac{h_\rho}{2}
			\right\}.
		\]
		}
	\end{block}

	\begin{itemize}
		\item Параметры $h_\rho$ и $h_\theta$ задают точность дискретизации пространства параметров.
			\item Для каждой активной точки $(x,y)$ и каждого значения $\theta_b$ вычисляется $\rho = x\cos\theta_b + y\sin\theta_b$, после чего увеличивается соответствующая ячейка аккумулятора.
		\item Поэтому локальные максимумы аккумуляторного массива интерпретируются как оценки параметров прямых.
	\end{itemize}
\end{frame}

\begin{frame}[shrink=18]{Преобразование Хафа: этапы работы}
	\textbf{Входные данные:} изображение после шумоподавления размера $m \times n$, порог $\varepsilon$, шаги дискретизации $h_{\rho}, h_{\theta}$, количество прямых $R$.

	\vspace{0.1cm}
	\textbf{Выходные данные:} оценки параметров найденных прямых $(\hat{\rho},\hat{\theta})$.

	\vspace{0.2cm}
	\begin{enumerate}
		\item Формируется активное множество $E_{\varepsilon}=\{(i,j):\widehat M_{ij}>\varepsilon\}$.
		\item Дискретизация пространства $(\rho,\theta)$ с шагами $h_\rho$ и $h_\theta$ и инициализация аккумулятора.
		\item Для каждой активной точки $(x,y)$ строится кривая $\rho(\theta)$ в пространстве параметров.
		\item Значения соответствующих ячеек аккумулятора увеличиваются на $1$.
		\item По аккумулятору ищутся $R$ локальных максимумов, после чего оцениваются параметры прямых.
		\item По найденным $(\hat{\rho},\hat{\theta})$ восстанавливаются прямые на изображении.
	\end{enumerate}

	\vspace{0.2cm}
	\begin{block}{Замечание}
		Качество предобработки напрямую влияет на точность финальных оценок $\rho$ и $\theta$.
	\end{block}
\end{frame}

\section{Алгоритмы на основе CSSA}

\begin{frame}{Стандартные методы: Median и Wiener}
	\scriptsize
	\begin{block}{Median}
		Пусть $W_{ij}^{(w)}$ --- квадратное окно размера $w\times w$ с центром в точке $(i,j)$ (стандартно $w=3$):
		\[
		\widehat M_{ij} =
		\operatorname{med}\{M_{pq}:(p,q)\in W_{ij}^{(w)}\}.
		\]
		Это обработка, подавляющая одиночные выбросы.
	\end{block}

	\begin{block}{Wiener}
		Пусть $W_{ij}^{(w)}$ --- квадратное окно размера $w\times w$ (стандартно $w=5$).
		По нему вычисляются локальные среднее $\mu_{ij}$ и дисперсия $\nu_{ij}$:
		\[
		\widehat M_{ij} =
		\mu_{ij}+\lambda_{ij}(M_{ij}-\mu_{ij}),\qquad
		\lambda_{ij}=
		\frac{\max(\nu_{ij}-\sigma_n^2,0)}{\max(\nu_{ij},\varepsilon)},
		\]
		где $\sigma_n^2$ --- оценка дисперсии шума, $\varepsilon>0$ предотвращает деление на нуль.
	\end{block}
\end{frame}

\begin{frame}{Стандартные методы: BM3D и Quantile $p$}
	\scriptsize
	\begin{block}{BM3D}
		Для опорного фрагмента $P_s$ в области поиска выбираются фрагменты $P_t$, близкие к нему по заданной мере сходства (например, MSE).
		Найденные фрагменты группируют, после чего применяется совместная фильтрация (например, Винеровским фильтром).
		После обратного преобразования перекрывающиеся оценки пикселей объединяются взвешенным усреднением.
	\end{block}

	\begin{block}{Quantile $p$}
		Для параметра $p\in(0,1)$ метод задает глобальный порог по $p$-квантилю яркостей:
		\[
		\widehat{M}_{ij}=M_{ij}\mathbf{1}\{M_{ij}\ge Q_{p}(\mathbf{M})\}.
		\]
		В численном эксперименте используется частный случай $p=0.9$.
	\end{block}
\end{frame}

\begin{frame}{CSSA: вложение комплексного ряда}
	\small
	\begin{itemize}
		\item Рассматривается комплексный временной ряд
		\[
		\mathsf{Z} = (Z_1,\dots,Z_N).
		\]
		\item Цель метода CSSA --- представить ряд как сумму сигнальной части $\widehat{\mathsf{S}}$ и шумовой части $\widehat{\mathsf{R}}$.
		\item Выбирается длина окна $L$; по умолчанию берется $L=(N+1)/2$. Далее задаётся $K = N - L + 1$.
		\item На вход CSSA подаются ряд $\mathsf{Z}$, длина окна $L$ и правило выбора сохраняемых компонент.
		\item Оператор вложения переводит ряд в траекторную матрицу
		\[
		\mathbf{Z} = \mathcal{T}_L(\mathsf{Z}) =
		\begin{pmatrix}
			Z_1 & Z_2 & \dots & Z_K \\
			Z_2 & Z_3 & \dots & Z_{K+1} \\
			\vdots & \vdots & \ddots & \vdots \\
			Z_L & Z_{L+1} & \dots & Z_N
		\end{pmatrix}.
		\]
	\end{itemize}
\end{frame}

\begin{frame}{CSSA: SVD, группировка и восстановление}
	\small
	\begin{itemize}
		\item Для траекторной матрицы строится сингулярное разложение
		$\mathbf{Z} = \sum_{a=1}^{d} \sqrt{\lambda_a}\,\mathbf{U}_a \mathbf{V}_a^{*}$.
		\item Компоненты, соответствующие сигналу, объединяются в группу $I$:
		$\mathbf{Z}_{I} = \sum_{a\in I} \sqrt{\lambda_a}\,\mathbf{U}_a \mathbf{V}_a^{*}$.
		\item Оператор $\mathcal{H}$ --- ганкелизация: усреднение элементов по побочным диагоналям траекторной матрицы.
		\item После диагонального усреднения вычисляется оценка комплексного ряда
		$\widehat{\mathsf{Z}} = \mathcal{T}_L^{-1}\circ\mathcal{H}(\mathbf{Z}_{I})$.
		\item Таким образом CSSA выступает как метод шумоподавления комплексного временного ряда.
	\end{itemize}

	\begin{block}{Применение в контексте шумоподавления изображений}
		Далее эта процедура используется внутри алгоритма \textbf{CSSA ROW-ROW}.
	\end{block}
\end{frame}

\begin{frame}{CSSA ROW-ROW: схема алгоритма}
	\small
	\textbf{Входные данные:} матрица изображения, $L$ --- размер окна, $r$ --- число сохраняемых компонент.

	\textbf{Выходные данные:} оценка изображения $\widehat{\mathbf{S}}\in\mathbb{R}^{m\times n}$.

	\vspace{0.2cm}
	\textbf{Схема работы:}
	\begin{enumerate}
		\item Для каждой строки $\mathsf{M}^{(p)}$ вычисляется дискретное преобразование Фурье (DFT):
		$\mathsf{Z}^{(p)} = \operatorname{DFT}\bigl(\mathsf{M}^{(p)}\bigr)$, $p=1,\dots,m$.
		\item К каждому комплексному ряду $\mathsf{Z}^{(p)}$ применяется $\operatorname{CSSA}(L,r)$.
		\item В разложении сохраняются $r$ компонент, после чего строится оценка ряда $\widehat{\mathsf{Z}}^{(p)}$.
		\item По каждой строке выполняется обратное преобразование Фурье (IDFT):
		$\widehat{\mathsf{S}}^{(p)} = \operatorname{Re}\!\Bigl(\operatorname{IDFT}\bigl(\widehat{\mathsf{Z}}^{(p)}\bigr)\Bigr)$.
		\item Из строк $\widehat{\mathsf{S}}^{(1)},\dots,\widehat{\mathsf{S}}^{(m)}$ составляется матрица $\widehat{\mathbf{S}}$.
	\end{enumerate}
\end{frame}

\begin{frame}[shrink=5]{Почему для одной точки ожидается ранг 1}
	\small
	Пусть в строке длины $N$ сигнал задается единичным ортом:
	\[
	\mathsf{X} = \mathsf{E}_j.
	\]
	После DFT получаем комплексную экспоненту:
	\[
	\operatorname{DFT}(\mathsf{E}_j)_t = \exp(i\omega_j(t-1)).
	\]

	\begin{block}{Связь с CSSA}
		Траекторная матрица такой экспоненты представляется как произведение:
		\[
		\mathbf{Z}_{ab} = \exp(i\omega_j(a+b-2)) = u_a v_b,
		\]
		где
		\[
		u_a=\exp(i\omega_j(a-1)),\qquad
		v_b=\exp(i\omega_j(b-1)).
		\]
		Следовательно, $\mathbf{Z}=uv^{T}$, причем $u\neq0$ и $v\neq0$, поэтому $\operatorname{rank}\mathbf{Z}=1$.
	\end{block}

	Поэтому чистая строка с одним пикселем сигнала после DFT является рядом комплексного ранга $1$ в смысле CSSA.
\end{frame}

\begin{frame}{Мотивировка алгоритма CSSA ROW-ROW}
	\small
	\begin{itemize}
		\item Для строки с одной яркой точкой и шумом имеем
		$\mathsf{X} = c\,\mathsf{E}_j + \boldsymbol{\varepsilon}$,
		где $\mathsf{E}_j$ --- базисный вектор, а $\boldsymbol{\varepsilon}$ --- гауссовский шум.
		\item После DFT
		$\operatorname{DFT}(\mathsf{X}) = c\,\mathsf{W}_j + \widetilde{\boldsymbol{\varepsilon}}$,
		где $\mathsf{W}_j=\operatorname{DFT}(\mathsf{E}_j)$ --- ряд из комплексных экспонент, имеющий ранг $1$, $\widetilde{\boldsymbol{\varepsilon}}$ --- гауссовский вектор в силу линейности DFT.
		\item В отсутствие шума траекторная матрица имеет ровно одну ненулевую сингулярную компоненту.
		\item При наличии шума лучшая одноранговая аппроксимация в норме Фробениуса задается первой компонентой SVD.
		\item Поэтому в \texttt{CSSA ROW-ROW} при одной активной точке в строке естественно сохранять одну компоненту.
	\end{itemize}
\end{frame}

\begin{frame}{Гауссовский шум после DFT}
	\scriptsize
	Пусть $\boldsymbol{\varepsilon}\sim \mathcal N(0,\sigma^2 I_N)$, а $F$ --- матрица DFT без дополнительной нормировки. Тогда
	\[
		\widetilde{\boldsymbol{\varepsilon}} = F\boldsymbol{\varepsilon}.
	\]

	\vspace{-0.2cm}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\textbf{Почему шум остается гауссовским}
			\[
				a^{*}\widetilde{\boldsymbol{\varepsilon}}
				= a^{*}F\boldsymbol{\varepsilon}
				= (F^{*}a)^{*}\boldsymbol{\varepsilon}.
			\]
			Это линейная комбинация координат гауссовского вектора, поэтому она гауссовская. Следовательно, $\widetilde{\boldsymbol{\varepsilon}}$ --- комплексный гауссовский вектор.
		\end{column}
		\begin{column}{0.48\textwidth}
			\textbf{Как меняется ковариация}
			\[
				F_{lm}=\exp\!\left(\frac{2\pi i}{N}(l-1)(m-1)\right),
			\]
				\[
					(FF^{*})_{lm}=
					\sum_{s=0}^{N-1}
					\exp\!\left(\frac{2\pi i}{N}s(l-m)\right).
				\]
			\[
				(FF^{*})_{lm}=N,\quad l=m,
			\]
			\vspace{-0.35cm}
			\[
				(FF^{*})_{lm}=0,\quad l\ne m.
			\]
		\end{column}
	\end{columns}

	\vspace{0.05cm}
	Следовательно,
	\[
		\operatorname{Cov}(F\boldsymbol{\varepsilon})=\sigma^2 FF^{*}=N\sigma^2 I_N.
	\]
\end{frame}

	\vspace{0.05cm}
	\begin{itemize}

		\item Шум делает ненулевыми последующие сингулярные значения.
		\item При меньшем шуме первая компонента доминирует сильнее, поэтому одноранговая реконструкция устойчивее.
	\end{itemize}
\end{frame}

\begin{frame}{Ограничения одноранговой построчной модели}
	\scriptsize
	Для $p$-й строки пусть $q_p$ --- число пикселей сигнала. После DFT:
	\[
	\mathcal F(\mathbf{x}^{(p)}) =
	\sum_{u=1}^{q_p} c_u\boldsymbol{\omega}_{j_u}
	+ \widetilde{\boldsymbol{\varepsilon}}^{(p)}.
	\]

	\vspace{-0.1cm}
	\begin{itemize}
		\item Ранг $1$ относится к случаю $q_p=1$, то есть к одной активной точке в строке, а не к одной прямой на всем изображении.
		\item При $q_p=0$ первая компонента CSSA выбирается по шуму.
		\item При $q_p\ge2$ часть сигнала уже не обязана быть одноранговой, поэтому одноранговая аппроксимация добавляет дополнительный источник ошибки.
		\item Поэтому число сохраняемых компонент $r_p$ должно выбираться внутри CSSA по сингулярному спектру или пороговому правилу; фиксированное $r_p=1$ является упрощением.
		\item Смещение максимума и растекание оценки могут возникать уже при $q_p=1$ из-за шума и возмущения первой SVD-компоненты.
	\end{itemize}
\end{frame}

\begin{frame}{Статистический выбор порога}
	\scriptsize
	Для одной строки после DFT имеем
	\[
		\mathsf{Z} = F\mathsf{X} = \mathsf{S}+\eta,
		\qquad
		\eta=F\boldsymbol{\varepsilon},
		\qquad
		\operatorname{Cov}(\eta)=N\sigma^2 I_N.
	\]
	После \texttt{CSSA ROW-ROW} первая компонента используется как оценка части сигнала:
	\[
		\widehat{\mathsf{S}},
		\qquad
		r=\mathsf{Z}-\widehat{\mathsf{S}}.
	\]
	Остаток $r$ переводится обратно в исходную область:
	\[
		\widehat{\boldsymbol{\varepsilon}}=F^{-1}r=\frac{1}{N}F^{*}r.
	\]
	Если $r$ содержит только шумовую часть, то
	\[
		\operatorname{Cov}(\widehat{\boldsymbol{\varepsilon}})
		=
		\frac{1}{N^2}F^{*}(N\sigma^2 I_N)F
		=
		\sigma^2 I_N,
	\]
	поскольку $F^{*}F=N I_N$.

	\vspace{-0.15cm}
	\begin{enumerate}
		\item оценить уровень остаточного шума по $\widehat{\boldsymbol{\varepsilon}}$, например через $\widehat\sigma=\operatorname{sd}(\widehat{\boldsymbol{\varepsilon}})$;
		\item выбрать порог $\tau$ как высокий квантиль распределения $|\widehat{\boldsymbol{\varepsilon}}|$;
		\item использовать $\varepsilon=\tau$ и занулить элементы $|\widehat M_{ij}|\le\tau$ перед преобразованием Хафа для всех методов предобработки.
	\end{enumerate}
\end{frame}

\begin{frame}{Итоги}
	\begin{itemize}
		\item Сравнение по конфигурациям показывает, что результаты не стоит агрегировать без разбора отдельных случаев.
		\item При $\sigma=0.2$ основной сбой \texttt{CSSA ROW-ROW} возникает на конфигурации ``крутая прямая''.
		\item При $\sigma=0.05$ этот сбой резко ослабевает, что указывает на существенную зависимость метода от уровня шума.
		\item \texttt{Quantile $p$} при $p=0.9$ остается наиболее стабильным в данном эксперименте, но его преимущество связано с конкретной моделью: одна прямая и одинаковая интенсивность сигнала.
	\end{itemize}
\end{frame}
