Вспомогательные функции

clip01 <- function(m) {
  m <- as.matrix(m)
  m[m < 0.001] <- 0
  m[m > 0.999] <- 1
  m
}

mse <- function(truth, pred) {
  mean((truth - pred)^2)
}

cssa_denoise <- function(m, num_of_lines = 1L, method = "row.row") {
  cleaned <- switch(
    method,
    "col.col" = m |> dft() |> cssa.col(num.line = num_of_lines) |> idft.col() |> Re(),
    "col.row" = m |> dft() |> cssa.col(num.line = num_of_lines) |> idft.row() |> Re(),
    "row.col" = m |> dft() |> cssa.row(num.line = num_of_lines) |> idft.col() |> Re(),
    "row.row" = m |> dft() |> cssa.row(num.line = num_of_lines) |> idft.row() |> Re(),
    stop("Unknown method: ", method)
  )
  clip01(cleaned)
}

line_coverage_stats <- function(m) {
  idx <- which(m > 0, arr.ind = TRUE)
  if (nrow(idx) == 0) {
    return(c(
      visible_pixels = 0,
      row_coverage = 0,
      col_coverage = 0
    ))
  }

  c(
    visible_pixels = nrow(idx),
    row_coverage = length(unique(idx[, 1])) / nrow(m),
    col_coverage = length(unique(idx[, 2])) / ncol(m)
  )
}

make_line_image <- function(n_row, n_col, a, b) {
  matrix(0, nrow = n_row, ncol = n_col) |>
    add.line(a = a, b = b)
}

row.row обходит другие алгоритмы в большинстве прямых. Точно по MSE он лучше, чем .col алгоритмы, так как они рассеивают сигнал на много кусков с одинаковым оглом наклона(вероятно, это происходит из-за idft по столбцу). Конечно, col.row тоже рассеивает сигнал(об этом в преобразовании Хафа), но частично это удается исправить с помощью увеличения num_of_lines(кол-ва компонент в SVD, которые мы считаем сигналом).

Функции для ошибки параметров HT

convert_ab_to_rho_theta <- function(a, b) {
  theta <- atan2(1, -a)
  rho <- b / sqrt(a^2 + 1)
  if (theta < 0) {
    theta <- theta + pi
    rho <- -rho
  }
  c(rho = rho, theta = theta)
}

find_k_max <- function(acc, k, qrho, qtheta, suppress = FALSE, window = 6) {
  acc_copy <- acc
  maxima <- vector("list", k)

  for (i in seq_len(k)) {
    max_val <- max(acc_copy)
    idx <- which(acc_copy == max_val, arr.ind = TRUE)[1, ]
    maxima[[i]] <- c(rho = qrho[idx[1]], theta = qtheta[idx[2]])

    if (suppress) {
      r_range <- max(1, idx[1] - window):min(nrow(acc_copy), idx[1] + window)
      c_range <- max(1, idx[2] - window):min(ncol(acc_copy), idx[2] + window)
      acc_copy[r_range, c_range] <- 0
    } else {
      acc_copy[idx[1], idx[2]] <- 0
    }
  }

  out <- do.call(rbind, maxima)
  colnames(out) <- c("rho", "theta")
  out
}

compute_err <- function(true, pred) {
  true <- as.matrix(true)
  pred <- as.matrix(pred)

  if (ncol(true) != 2 || ncol(pred) != 2) {
    stop("true and pred must have two columns: rho and theta")
  }

  if (nrow(true) != nrow(pred)) {
    stop("true and pred must have the same number of lines")
  }

  pair_err <- function(tr, tt, pr, pt) {
    pt_alt <- (pt + pi) %% (2 * pi)
    dtheta <- atan2(sin(tt - pt), cos(tt - pt))^2
    dtheta_alt <- atan2(sin(tt - pt_alt), cos(tt - pt_alt))^2

    loss_1 <- (tr - pr)^2 + dtheta
    loss_2 <- (tr + pr)^2 + dtheta_alt

    if (loss_1 <= loss_2) {
      c(dr = (tr - pr)^2, dtheta = dtheta, total = loss_1)
    } else {
      c(dr = (tr + pr)^2, dtheta = dtheta_alt, total = loss_2)
    }
  }

  n_lines <- nrow(true)
  cost <- matrix(0, nrow = n_lines, ncol = n_lines)

  for (i in seq_len(n_lines)) {
    for (j in seq_len(n_lines)) {
      cost[i, j] <- pair_err(
        true[i, 1], true[i, 2],
        pred[j, 1], pred[j, 2]
      )[3]
    }
  }

  if (n_lines == 1) {
    out <- pair_err(true[1, 1], true[1, 2], pred[1, 1], pred[1, 2])
    return(c(dr = unname(out[1]), dtheta = unname(out[2])))
  }

  all_perms <- function(v) {
    if (length(v) == 1) {
      return(matrix(v, nrow = 1))
    }

    out <- lapply(seq_along(v), function(i) {
      rest <- all_perms(v[-i])
      cbind(v[i], rest)
    })
    do.call(rbind, out)
  }

  best_perm <- NULL
  best_cost <- Inf
  perm_mat <- all_perms(seq_len(n_lines))
  for (row_i in seq_len(nrow(perm_mat))) {
    cur_perm <- perm_mat[row_i, ]
    cur_cost <- sum(cost[cbind(seq_len(n_lines), cur_perm)])
    if (cur_cost < best_cost) {
      best_cost <- cur_cost
      best_perm <- cur_perm
    }
  }

  dr_total <- 0
  dtheta_total <- 0
  for (i in seq_len(n_lines)) {
    out <- pair_err(
      true[i, 1], true[i, 2],
      pred[best_perm[i], 1], pred[best_perm[i], 2]
    )
    dr_total <- dr_total + out[1]
    dtheta_total <- dtheta_total + out[2]
  }

  c(dr = unname(dr_total), dtheta = unname(dtheta_total))
}

      method_name <- methods_ht[method_i]
      detector_fun <- detectors_ht[[method_i]]
      
      res <- make_accumulator(
        noisy_matrix,
        detector_fun,
        rho_step = rho_step_ht,
        theta_step = theta_step_ht
      )

      pred_plain <- find_k_max(
        res$accumulator,
        k = num_of_lines,
        qrho = res$rho,
        qtheta = res$theta,
        suppress = FALSE,
        window = 0
      )
      
    
      
      err_plain <- compute_err(true_line, pred_plain)

      ht_rows[[ht_idx]] <- data.frame(
        config_id = cfg$config_id,
        description = cfg$description,
        n_col = cfg$n_col,
        n_row = cfg$n_row,
        a = cfg$a,
        b = cfg$b,
        sigma = sigma_noise,
        rep = rep_i,
        method = method_name,
        true_rho = true_line[1, 1],
        true_theta = true_line[1, 2],
        pred_rho = pred_plain[1, 1],
        pred_theta = pred_plain[1, 2],
        dr = as.numeric(err_plain[1]),
        dtheta = as.numeric(err_plain[2]),
        stringsAsFactors = FALSE
      )
      ht_idx <- ht_idx + 1L
    }
  }
  
}
